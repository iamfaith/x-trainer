1.
    记住拷贝你需要的checkoutpoints出来，譬如我要这一个：
    cp -r ./outputs/train/act_task1_50_chunk900_action200/checkpoints/022000/pretrained_model ./

2. 创建Dockerfile，输入下面内容：


FROM gdpi_isaac-lab
# 或使用你已配好的 lerobot 环境镜像

WORKDIR /workspace

RUN /workspace/isaaclab/_isaac_sim/python.sh -m pip install lerobot -i https://mirrors.tuna.tsinghua.edu.cn/pypi/web/simple

# 将本地训练好的权重拷贝到镜像内固定路径
COPY ./pretrained_model /workspace/checkpoints/pretrained_model

# 暴露推理端口
EXPOSE 5555

# 启动推理服务，监听 0.0.0.0
CMD ["/workspace/isaaclab/_isaac_sim/python.sh", "-m", "lerobot.scripts.server.policy_server", "--host=0.0.0.0", "--port=5555", "--fps=30"]



3. 在Dockerfile目录下运行


# 在 x-trainer 或你的项目目录构建，将 0001 替换为你的四位数队伍编号
docker build -t dobot_0001_submission:v1 -f Dockerfile .

# 运行容器并映射 5555 端口（自测用，Tag 如 v1 表示第一次提交）
docker run -it --rm -p 5555:5555 --gpus all dobot_0001_submission:v1



docker run --rm -it --entrypoint bash  -p 5555:5555 --gpus all dobot_0001_submission:v1
# 然后在容器内运行：
which python || which python3
python3 --version || python --version
ls -l /usr/bin | grep python
env | grep PATH

docker run -it --rm -p 5555:5555 --gpus all dobot_0001_submission:v1 \
  python3 -m lerobot.scripts.server --host=0.0.0.0 --port=5555 --fps=30
